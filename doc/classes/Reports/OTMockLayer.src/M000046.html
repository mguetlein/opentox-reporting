  def init_predictions(predictions, test_dataset_uri, prediction_dataset_uri)
  
    p = Array.new
    c = Array.new
    u = Array.new
    (0..NUM_PREDICTIONS).each do |i|
      p.push rand
      c.push rand(2)
      u.push((i+1))
    end
    predictions.predicted_values = p
    predictions.actual_values = c
    predictions.compound_uris = u
  end
