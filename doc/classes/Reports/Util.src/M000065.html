  def self.group( objects, equal_attributes )
    
    result = Array.new
    
    objects.each do |o|
      
      match = false
      result.each do |r|
        if o.has_equal_attributes?(equal_attributes, r[0])
          match = true
          r.push(o)
          break
        end
      end
      
      if !match
        result.push(Array.new)
        result[-1].push(o)
      end
    end
    
    return result    
  end
